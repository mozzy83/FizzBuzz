The Code for Fizz Buzz.

                            
//Get values from user
function getValues (){
    let fizzValue = document.getElementById('fizzValue').value;
    let buzzValue = document.getElementById('buzzValue').value;
    //parse into integers
    fizzValue = parseInt(fizzValue);
    buzzValue = parseInt(buzzValue);
    //validation
    if(Number.isInteger(fizzValue) && Number.isInteger(buzzValue)){
    //call fizzBuzz
    let theArray = fizzBuzz(fizzValue, buzzValue);
    //call displayResults
    displayResults(theArray);
    }
    else{alert('You must enter an integer');}
}
//Generate an array setting fizzbuzz values
function fizzBuzz(fizzValue, buzzValue) {
    let theArray = [];
    for (let i = 1; i <= 100; i++){
        if((i % fizzValue == 0) && (i % buzzValue == 0)) {theArray.push('FizzBuzz');}
        else if(i % fizzValue == 0) {theArray.push('Fizz');}
        else if(i % buzzValue == 0) {theArray.push('Buzz');}
        else {theArray.push(i);}
    }
    return theArray;
}
//Display Results to User
function displayResults(theArray) {
    //get table body element
    let tableBody = document.getElementById("resultsTable");
    //get the template row
    let templateRow = document.getElementById("fbTemplate");
    //clear table first
    tableBody.innerHTML = '';
    for (let i = 0; i < theArray.length; i += 5) {
        let tableRow = document.importNode(templateRow.content, true);
        //grab just the td's and put into array
        let rowCols = tableRow.querySelectorAll('td');
        rowCols[0].classList.add(theArray[i]);
        rowCols[0].textContent = theArray[i];
        rowCols[1].classList.add(theArray[i+1]);
        rowCols[1].textContent = theArray[i+1];
        rowCols[2].classList.add(theArray[i+2]);
        rowCols[2].textContent = theArray[i+2];
        rowCols[3].classList.add(theArray[i+3]);
        rowCols[3].textContent = theArray[i+3];
        rowCols[4].classList.add(theArray[i+4]);
        rowCols[4].textContent = theArray[i+4];

        tableBody.appendChild(tableRow);
    }
}

Fizz Buzz is structured in three main functions:

getValues()

The initial function retrieves the Fizz and Buzz values entered by the user and parses them into integers. It then performs a simple validation to make sure both the values parsed properly. If the values passed validation they are sent to the next function, fizzBuzz...

fizzBuzz()

This function initially declares an array and runs a For Loop to populate it. As it puts each number into the array it checks it to see if it is a multiple of both of the user's values, a multiple of the Fizz value, and finally a multiple of the Buzz value. If it matches any of these criteria it is put into the array as the appropriate string. If it matches none of them it is put in as the number. Once it has looped through 100 times the function returns the array that was built so that it can be passed to the last function, displayResults...

displayResults()

The last function grabs a handle to the empty table we created in the HTML file. It then grabs a handle to the template that we also created. Before beginning to populate our table it makes sure the table is clear. The function then runs a For Loop where it grabs a handle to our template, creates an array of just the td elements inside the template, adds the first 5 values from our original array as the "textContent" of the td elements (which are the table columns), and finally appends those 5 columns to the table. It then loops through with the next 5 values of our array.

Styling is applied to each td column using the classList.add() method and using the current value of theArray as the td's class value. Three ID tags were added to the css file; Fizz, Buzz, and FizzBuzz. When the current value of theArray is a number there is no styling, but when the value matches a tag in the style sheet the appropriate styling is applied.
